"use strict";

import { indexPage } from "./pages/index-page.js";
import { notFoundPage } from "./pages/not-found-page.js";

/**
 * @file SPA Page Handler
 * @module PageHandler
 * @description
 * This module manages the Single Page Application (SPA) logic.<br>
 * Key features:<br>
 * 1. Automatically imports page modules using Vite's glob import.<br>
 * 2. Manages dynamic content rendering without page reloads.<br>
 * 3. Synchronizes with the Browser History API (pushState/popstate).
 */

// Automatically imports all page files matching the pattern
const pageFiles = import.meta.glob("./pages/*-page.js", { eager: true });

/**
 * Initializes the page handling logic.
 * @function initPageHandler
 */
export function initPageHandler() {
  /** 
   * The main container where page content is injected.
   * @type {HTMLElement} 
  */
  const app = document.querySelector('#app');

  /** 
   * Mapping of page keys to their HTML content.
   * @type {Object<string, string>}
   */
  const pages = {
    'index': indexPage,
    'not-found': notFoundPage,
  };

  // 1. Populate the pages object dynamically from imported files
  for (const path in pageFiles) {
    const pageKey = path.split('/').pop().replace('-page.js', '');
    const pageKeyValue = `${pageKey}Page`;
    pages[pageKey] = pageFiles[path][pageKeyValue];
  }

  /**
   * Renders the HTML content of the selected page and updates the browser URL.
   * @param {string} activePage - The key of the page to be displayed.
   * @param {boolean} [updateHistory=true] - Whether to push a new state to the browser history.
   * @function renderPageContent
   * @inner
   */
  const renderPageContent = (activePage, updateHistory = true) => {
    // 2. Updates content and scrolls window to top
    app.innerHTML = pages[activePage] || notFoundPage;
    window.scrollTo(0, 0);

    // 3. Adds history for browser
    if (updateHistory) {
      const url = activePage === 'index' ? '/' : `/${activePage}`;
      window.history.pushState({ page: activePage }, "", url);
    }

    // 4. Updates active page class
    updateActivePageClass(activePage);
  };

  /**
   * Updates the 'active-page' CSS class on navigation links.
   * @param {string} activePage - The key of the currently active page.
   * @function updateActivePageClass
   * @inner
   */
  function updateActivePageClass(activePage) {
    const navLinks = document.querySelectorAll('.nav-list a');
    navLinks.forEach(link => {
      if (link.getAttribute('data-page') === activePage) {
        link.classList.add('active-page');
      } else {
        link.classList.remove('active-page');
      }
    });
  }

  // 5. Listens to browser back/forward buttons (popstate)
  window.addEventListener('popstate', (event) => {
    let pageToLoad = event.state && event.state.page;
    if (!pageToLoad) {
      pageToLoad = window.location.pathname.replace('/', '') || 'index';
    }
    renderPageContent(pageToLoad, false);
  });

  // 6. Intercepts clicks on elements with the [data-page] attribute
  document.addEventListener('click', (event) => {
    const link = event.target.closest('[data-page]');
    if (link) {
      event.preventDefault();
      const activePage = link.getAttribute('data-page');
      renderPageContent(activePage);
    }
  });

  // 7. Initial render based on current URL path on load
  const currentPath = window.location.pathname.replace('/', '');
  const initialPage = currentPath || 'index';
  renderPageContent(initialPage, false);
}
